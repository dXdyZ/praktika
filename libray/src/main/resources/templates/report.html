Задание 1

Преобразовать строки матрицы таким образом, чтобы элементы, равные нулю, располагались после всех остальных.

Код:
public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Введите размерность матрицы n: ");
        int n = scanner.nextInt();
        int[][] matrix = new int[n][n];
        Random random = new Random();

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                matrix[i][j] = random.nextInt(2 * n + 1) - n;
            }
        }

        System.out.println("Исходная матрица:");
        printMatrix(matrix);

        for (int i = 0; i < n; i++) {
            matrix[i] = transformRow(matrix[i]);
        }

        System.out.println("Матрица после преобразования:");
        printMatrix(matrix);
    }

    private static int[] transformRow(int[] row) {
        int[] result = new int[row.length];
        int index = 0;
        for (int value : row) {
            if (value != 0) {
                result[index++] = value;
            }
        }
        return result;
    }

    private static void printMatrix(int[][] matrix) {
        for (int[] row : matrix) {
            System.out.println(Arrays.toString(row));
        }
    }
}

Задание 2

Реализовать класс Fraction для работы с дробями: сложение, вычитание, умножение и деление дробей.

Код:
class Fraction {
    private int numerator;
    private int denominator;

    public Fraction(int numerator, int denominator) {
        if (denominator == 0) throw new IllegalArgumentException("Знаменатель не может быть равен 0");
        this.numerator = numerator;
        this.denominator = denominator;
        simplify();
    }

    public Fraction add(Fraction other) {
        int newNumerator = this.numerator * other.denominator + other.numerator * this.denominator;
        int newDenominator = this.denominator * other.denominator;
        return new Fraction(newNumerator, newDenominator);
    }

    public Fraction subtract(Fraction other) {
        int newNumerator = this.numerator * other.denominator - other.numerator * this.denominator;
        int newDenominator = this.denominator * other.denominator;
        return new Fraction(newNumerator, newDenominator);
    }

    public Fraction multiply(Fraction other) {
        return new Fraction(this.numerator * other.numerator, this.denominator * other.denominator);
    }

    public Fraction divide(Fraction other) {
        return new Fraction(this.numerator * other.denominator, this.denominator * other.numerator);
    }

    private void simplify() {
        int gcd = gcd(numerator, denominator);
        numerator /= gcd;
        denominator /= gcd;
    }

    private int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return Math.abs(a);
    }

    @Override
    public String toString() {
        return numerator + "/" + denominator;
    }
}

Задание 4

Создать класс для работы с отделами магазина, добавлять товары и услуги в отделы, а также отображать содержимое каждого отдела.

Код:
import java.util.ArrayList;
import java.util.List;

class Shop {
    private List departments;

    public Shop() {
        departments = new ArrayList<>();
    }

    public void addDepartment(String name) {
        departments.add(new Department(name));
    }

    public void addProductToDepartment(String departmentName, String product) {
        for (Department department : departments) {
            if (department.getName().equals(departmentName)) {
                department.addProduct(product);
                return;
            }
        }
    }

    public void displayDepartments() {
        for (Department department : departments) {
            System.out.println(department);
        }
    }

    class Department {
        private String name;
        private List products;
        private List services;

        public Department(String name) {
            this.name = name;
            products = new ArrayList<>();
            services = new ArrayList<>();
        }

        public String getName() {
            return name;
        }

        public void addProduct(String product) {
            products.add(product);
        }

        public void addService(String service) {
            services.add(service);
        }

        @Override
        public String toString() {
            return "Отдел: " + name + "\\nТовары: " + products + "\\nУслуги: " + services + "\\n";
        }
    }
}

Задание 5

Найти уникальные слова в тексте, у которых первая и последняя буквы совпадают.

Код:
import java.util.HashSet;
import java.util.Scanner;
import java.util.Set;

public class UniqueWords {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.println("Введите текст:");
        String text = scanner.nextLine();

        Set uniqueWords = new HashSet<>();
        for (String word : text.split("\\s+")) {
            if (word.length() > 1 && word.charAt(0) == word.charAt(word.length() - 1)) {
                uniqueWords.add(word);
            }
        }

        System.out.println("Уникальные слова с совпадающими первой и последней буквами: " + uniqueWords);
    }
}

Задание 9

Реализовать итальянскую игру «Математико» с использованием колоды карт и игрового поля.

Код:
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

class Card {
    private final String suit;
    private final String rank;

    public Card(String suit, String rank) {
        this.suit = suit;
        this.rank = rank;
    }

    @Override
    public String toString() {
        return rank + " of " + suit;
    }
}

class Deck {
    private final List cards;

    public Deck() {
        String[] suits = {"Hearts", "Diamonds", "Clubs", "Spades"};
        String[] ranks = {"2", "3", "4", "5", "6", "7", "8", "9", "10", "Jack", "Queen", "King", "Ace"};
        cards = new ArrayList<>();

        for (String suit : suits) {
            for (String rank : ranks) {
                cards.add(new Card(suit, rank));
            }
        }
        Collections.shuffle(cards);
    }

    public Card drawCard() {
        return cards.remove(cards.size() - 1);
    }
}

public class MatematikoGame {
    public static void main(String[] args) {
        Deck deck = new Deck();
        System.out.println("Начнем игру «Математико»!");

        // Example of drawing a card
        Card drawnCard = deck.drawCard();
        System.out.println("Вытащена карта: " + drawnCard);
    }
}
